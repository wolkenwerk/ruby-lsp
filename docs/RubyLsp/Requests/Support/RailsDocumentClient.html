def generate_rails_document_urls(name)
  docs = search_index&.fetch(name, nil)

  return [] unless docs

  docs.map do |doc|
    owner = doc[:owner]

    link_name =
      # class/module name
      if owner == name
        name
      else
        "#{owner}##{name}"
      end

    "[Rails Document: `#{link_name}`](#{doc[:url]})"
  end
end
